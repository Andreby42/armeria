103     /**
104      * Sets graceful shutdown timeout, health check services and {@link MeterRegistry} for the specified
105      * {@link ServerBuilder}.
106      */
107     public static void configureServerWithArmeriaSettings(ServerBuilder server, ArmeriaSettings settings,
108                                                           MeterRegistry meterRegistry,
109                                                           List<HealthChecker> healthCheckers) {
110         requireNonNull(server, "server");
111         requireNonNull(settings, "settings");
112         requireNonNull(meterRegistry, "meterRegistry");
113         requireNonNull(healthCheckers, "healthCheckers");
114 
115         if (settings.getGracefulShutdownQuietPeriodMillis() >= 0 &&
116             settings.getGracefulShutdownTimeoutMillis() >= 0) {
117             server.gracefulShutdownTimeout(settings.getGracefulShutdownQuietPeriodMillis(),
118                                            settings.getGracefulShutdownTimeoutMillis());
119             logger.debug("Set graceful shutdown timeout: quiet period {} ms, timeout {} ms",
120                          settings.getGracefulShutdownQuietPeriodMillis(),
121                          settings.getGracefulShutdownTimeoutMillis());
122         }
123 
124         final String healthCheckPath = settings.getHealthCheckPath();
125         if (!Strings.isNullOrEmpty(healthCheckPath)) {
126             server.service(healthCheckPath,
127                            new HttpHealthCheckService(healthCheckers.toArray(EMPTY_HEALTH_CHECKERS)));
128         }
129 
130         server.meterRegistry(meterRegistry);
131 
132         if (settings.isEnableMetrics() && !Strings.isNullOrEmpty(settings.getMetricsPath())) {
133             if (meterRegistry instanceof CompositeMeterRegistry) {
134                 final Set<MeterRegistry> childRegistries =
135                         ((CompositeMeterRegistry) meterRegistry).getRegistries();
136                 childRegistries.stream()
137                                .filter(PrometheusMeterRegistry.class::isInstance)
138                                .map(PrometheusMeterRegistry.class::cast)
139                                .findAny()
140                                .ifPresent(r -> addPrometheusExposition(settings, server, r));
141             } else if (meterRegistry instanceof PrometheusMeterRegistry) {
142                 addPrometheusExposition(settings, server, (PrometheusMeterRegistry) meterRegistry);
143             } else if (meterRegistry instanceof DropwizardMeterRegistry) {
144                 final MetricRegistry dropwizardRegistry =
145                         ((DropwizardMeterRegistry) meterRegistry).getDropwizardRegistry();
146                 final ObjectMapper objectMapper = new ObjectMapper()
147                         .enable(SerializationFeature.INDENT_OUTPUT)
148                         .registerModule(new MetricsModule(TimeUnit.SECONDS, TimeUnit.MILLISECONDS, true));
149                 server.service(
150                         settings.getMetricsPath(),
151                         new AbstractHttpService() {
152                             @Override
153                             protected HttpResponse doGet(ServiceRequestContext ctx, HttpRequest req)
154                                     throws Exception {
155                                 return HttpResponse.of(HttpStatus.OK, MediaType.JSON_UTF_8,
156                                                        objectMapper.writeValueAsBytes(dropwizardRegistry));
157                             }
158                         });
159             }
160         }
161 
162         if (settings.getSsl() != null) {
163             configureTls(server, settings.getSsl());
164         }
165     }

167     /**
168      * Adds {@link Port}s to the specified {@link ServerBuilder}.
169      */
170     public static void configurePorts(ServerBuilder server, List<Port> ports) {
171         requireNonNull(server, "server");
172         requireNonNull(ports, "ports");
173         ports.forEach(p -> {
174             final String ip = p.getIp();
175             final String iface = p.getIface();
176             final int port = p.getPort();
177             final List<SessionProtocol> protocols = p.getProtocols();
178 
179             if (ip == null) {
180                 if (iface == null) {
181                     server.port(new ServerPort(port, protocols));
182                 } else {
183                     try {
184                         final Enumeration<InetAddress> e = NetworkInterface.getByName(iface).getInetAddresses();
185                         while (e.hasMoreElements()) {
186                             server.port(new ServerPort(new InetSocketAddress(e.nextElement(), port),
187                                                        protocols));
188                         }
189                     } catch (SocketException e) {
190                         throw new IllegalStateException("Failed to find an iface: " + iface, e);
191                     }
192                 }
193             } else if (iface == null) {
194                 if (NetUtil.isValidIpV4Address(ip) || NetUtil.isValidIpV6Address(ip)) {
195                     final byte[] bytes = NetUtil.createByteArrayFromIpAddressString(ip);
196                     try {
197                         server.port(new ServerPort(new InetSocketAddress(
198                                 InetAddress.getByAddress(bytes), port), protocols));
199                     } catch (UnknownHostException e) {
200                         // Should never happen.
201                         throw new Error(e);
202                     }
203                 } else {
204                     throw new IllegalStateException("invalid IP address: " + ip);
205                 }
206             } else {
207                 throw new IllegalStateException("A port cannot have both IP and iface: " + p);
208             }
209         });
210     }

212     /**
213      * Adds Thrift services to the specified {@link ServerBuilder}.
214      */
215     public static void configureThriftServices(
216             ServerBuilder server, List<ThriftServiceRegistrationBean> beans,
217             @Nullable MeterIdPrefixFunctionFactory meterIdPrefixFunctionFactory,
218             @Nullable String docsPath) {
219         requireNonNull(server, "server");
220         requireNonNull(beans, "beans");
221 
222         final List<TBase<?, ?>> docServiceRequests = new ArrayList<>();
223         final Map<String, Collection<HttpHeaders>> docServiceHeaders = new HashMap<>();
224         beans.forEach(bean -> {
225             Service<HttpRequest, HttpResponse> service = bean.getService();
226             for (Function<Service<HttpRequest, HttpResponse>, ? extends Service<HttpRequest, HttpResponse>>
227                     decorator : bean.getDecorators()) {
228                 service = service.decorate(decorator);
229             }
230             service = setupMetricCollectingService(service, bean, meterIdPrefixFunctionFactory);
231             server.service(bean.getPath(), service);
232             docServiceRequests.addAll(bean.getExampleRequests());
233             ThriftServiceUtils.serviceNames(bean.getService())
234                               .forEach(serviceName -> docServiceHeaders
235                                       .put(serviceName,
236                                            bean.getExampleHeaders()));
237         });
238 
239         if (!Strings.isNullOrEmpty(docsPath)) {
240             final DocServiceBuilder docServiceBuilder = new DocServiceBuilder();
241             docServiceBuilder.exampleRequest(docServiceRequests);
242             for (Entry<String, Collection<HttpHeaders>> entry : docServiceHeaders.entrySet()) {
243                 docServiceBuilder.exampleHttpHeaders(entry.getKey(), entry.getValue());
244             }
245             server.serviceUnder(docsPath, docServiceBuilder.build());
246         }
247     }

250      * Adds HTTP services to the specified {@link ServerBuilder}.
251      */
252     public static void configureHttpServices(
253             ServerBuilder server, List<HttpServiceRegistrationBean> beans,
254             @Nullable MeterIdPrefixFunctionFactory meterIdPrefixFunctionFactory) {
255         requireNonNull(server, "server");
256         requireNonNull(beans, "beans");
257 
258         beans.forEach(bean -> {
259             Service<HttpRequest, HttpResponse> service = bean.getService();
260             for (Function<Service<HttpRequest, HttpResponse>, ? extends Service<HttpRequest, HttpResponse>>
261                     decorator : bean.getDecorators()) {
262                 service = service.decorate(decorator);
263             }
264             service = setupMetricCollectingService(service, bean, meterIdPrefixFunctionFactory);
265             server.service(bean.getPathMapping(), service);
266         });
267     }

269     /**
270      * Adds annotated HTTP services to the specified {@link ServerBuilder}.
271      */
272     public static void configureAnnotatedHttpServices(
273             ServerBuilder server, List<AnnotatedServiceRegistrationBean> beans,
274             @Nullable MeterIdPrefixFunctionFactory meterIdPrefixFunctionFactory) {
275         requireNonNull(server, "server");
276         requireNonNull(beans, "beans");
277 
278         beans.forEach(bean -> {
279             Function<Service<HttpRequest, HttpResponse>,
280                     ? extends Service<HttpRequest, HttpResponse>> decorator = Function.identity();
281             for (Function<Service<HttpRequest, HttpResponse>, ? extends Service<HttpRequest, HttpResponse>>
282                     d : bean.getDecorators()) {
283                 decorator = decorator.andThen(d);
284             }
285             if (meterIdPrefixFunctionFactory != null) {
286                 decorator = decorator.andThen(
287                         metricCollectingServiceDecorator(bean, meterIdPrefixFunctionFactory));
288             }
289             final ImmutableList<Object> exceptionHandlersAndConverters =
290                     ImmutableList.builder()
291                                  .addAll(bean.getExceptionHandlers())
292                                  .addAll(bean.getRequestConverters())
293                                  .addAll(bean.getResponseConverters())
294                                  .build();
295             server.annotatedService(bean.getPathPrefix(), bean.getService(), decorator,
296                                     exceptionHandlersAndConverters);
297         });
298     }

300     private static Service<HttpRequest, HttpResponse> setupMetricCollectingService(
301             Service<HttpRequest, HttpResponse> service,
302             AbstractServiceRegistrationBean<?, ?> bean,
303             @Nullable MeterIdPrefixFunctionFactory meterIdPrefixFunctionFactory) {
304         requireNonNull(service, "service");
305         requireNonNull(bean, "bean");
306 
307         if (meterIdPrefixFunctionFactory == null) {
308             return service;
309         }
310         return service.decorate(metricCollectingServiceDecorator(bean, meterIdPrefixFunctionFactory));
311     }

313     private static Function<Service<HttpRequest, HttpResponse>,
314             MetricCollectingService<HttpRequest, HttpResponse>> metricCollectingServiceDecorator(
315             AbstractServiceRegistrationBean<?, ?> bean,
316             MeterIdPrefixFunctionFactory meterIdPrefixFunctionFactory) {
317         requireNonNull(bean, "bean");
318         requireNonNull(meterIdPrefixFunctionFactory, "meterIdPrefixFunctionFactory");
319 
320         return MetricCollectingService.newDecorator(
321                 meterIdPrefixFunctionFactory.get(METER_TYPE, bean.getServiceName()));
322     }

324     private static void addPrometheusExposition(ArmeriaSettings armeriaSettings, ServerBuilder server,
325                                                 PrometheusMeterRegistry registry) {
326         requireNonNull(armeriaSettings, "armeriaSettings");
327         requireNonNull(server, "server");
328         requireNonNull(registry, "registry");
329 
330         final String metricsPath = armeriaSettings.getMetricsPath();
331         if (metricsPath == null) {
332             return;
333         }
334 
335         final CollectorRegistry prometheusRegistry = registry.getPrometheusRegistry();
336         server.service(metricsPath, new PrometheusExpositionService(prometheusRegistry));
337     }

346     /**
347      * Adds SSL/TLS context to the specified {@link ServerBuilder}.
348      */
349     public static void configureTls(ServerBuilder sb, Ssl ssl,
350                                     @Nullable Supplier<KeyStore> keyStoreSupplier,
351                                     @Nullable Supplier<KeyStore> trustStoreSupplier) {
352         if (!ssl.isEnabled()) {
353             return;
354         }
355         try {
356             if (keyStoreSupplier == null && trustStoreSupplier == null &&
357                 ssl.getKeyStore() == null && ssl.getTrustStore() == null) {
358                 logger.warn("Configuring TLS with a self-signed certificate " +
359                             "because no key or trust store was specified");
360                 sb.tlsSelfSigned();
361                 return;
362             }
363 
364             final SslContextBuilder sslBuilder = SslContextBuilder
365                     .forServer(getKeyManagerFactory(ssl, keyStoreSupplier))
366                     .trustManager(getTrustManagerFactory(ssl, trustStoreSupplier));
367 
368             final List<String> enabledProtocols = ssl.getEnabledProtocols();
369             if (enabledProtocols != null) {
370                 sslBuilder.protocols(enabledProtocols.toArray(new String[enabledProtocols.size()]));
371             }
372 
373             final List<String> ciphers = ssl.getCiphers();
374             if (ciphers != null) {
375                 sslBuilder.ciphers(ImmutableList.copyOf(ciphers));
376             }
377 
378             final ClientAuth clientAuth = ssl.getClientAuth();
379             if (clientAuth != null) {
380                 sslBuilder.clientAuth(clientAuth);
381             }
382 
383             sb.tls(sslBuilder.build());
384         } catch (Exception e) {
385             throw new IllegalStateException("Failed to configure TLS: " + e, e);
386         }
387     }

389     private static KeyManagerFactory getKeyManagerFactory(
390             Ssl ssl, @Nullable Supplier<KeyStore> sslStoreProvider) throws Exception {
391         final KeyStore store;
392         if (sslStoreProvider != null) {
393             store = sslStoreProvider.get();
394         } else {
395             store = loadKeyStore(ssl.getKeyStoreType(), ssl.getKeyStore(), ssl.getKeyStorePassword());
396         }
397 
398         final KeyManagerFactory keyManagerFactory =
399                 KeyManagerFactory.getInstance(KeyManagerFactory.getDefaultAlgorithm());
400 
401         String keyPassword = ssl.getKeyPassword();
402         if (keyPassword == null) {
403             keyPassword = ssl.getKeyStorePassword();
404         }
405 
406         keyManagerFactory.init(store, keyPassword != null ? keyPassword.toCharArray()
407                                                           : null);
408         return keyManagerFactory;
409     }
410 
411     private static TrustManagerFactory getTrustManagerFactory(
412             Ssl ssl, @Nullable Supplier<KeyStore> sslStoreProvider) throws Exception {
413         final KeyStore store;
414         if (sslStoreProvider != null) {
415             store = sslStoreProvider.get();
416         } else {
417             store = loadKeyStore(ssl.getTrustStoreType(), ssl.getTrustStore(), ssl.getTrustStorePassword());
418         }
419 
420         final TrustManagerFactory trustManagerFactory =
421                 TrustManagerFactory.getInstance(TrustManagerFactory.getDefaultAlgorithm());
422         trustManagerFactory.init(store);
423         return trustManagerFactory;
424     }
425 
426     @Nullable
427     private static KeyStore loadKeyStore(
428             @Nullable String type,
429             @Nullable String resource,
430             @Nullable String password) throws IOException, GeneralSecurityException {
431         if (resource == null) {
432             return null;
433         }
434         final KeyStore store = KeyStore.getInstance(firstNonNull(type, "JKS"));
435         final URL url = ResourceUtils.getURL(resource);
436         store.load(url.openStream(), password != null ? password.toCharArray()
437                                                       : null);
438         return store;
439     }
